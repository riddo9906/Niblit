# requirements/android.txt  —  Android / proot-Ubuntu profile
# ═══════════════════════════════════════════════════════════════════════════════
# A strict subset of core.txt that avoids packages known to fail on ARM proot:
#
#   ✗  faiss-cpu          — requires libopenblas; painful to build on ARM
#   ✗  sentence-transformers — pulls torch (800 MB+)
#   ✗  transformers        — large; needed only for local Qwen training
#   ✗  accelerate          — training/GPU harness; not needed on Android
#   ✗  sentencepiece       — native extension, often fails in proot
#   ✗  safetensors         — needed only with transformers backend
#   ✗  scrapy              — heavy crawler; prefer requests+bs4 path
#   ✗  docker              — no daemon available in proot
#   ✗  lean / QuantConnect CLI — requires Lean4 binary
#   ✗  scikit-learn        — large; only needed for advanced clustering
#
# Niblit will start with NIBLIT_PROFILE=android and use:
#   • numpy cosine-similarity as vector backend  (NIBLIT_VECTOR_BACKEND=numpy)
#   • remote embeddings or "none"                 (NIBLIT_EMBEDDINGS_BACKEND=none)
#
# Install:
#   pip install -r requirements/android.txt
#   export NIBLIT_PROFILE=android
# ═══════════════════════════════════════════════════════════════════════════════
